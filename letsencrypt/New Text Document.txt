letsencrypt-prod.yml
====================
apiVersion: cert-manager.io/v1
kind: ClusterIssuer
metadata:
  name: letsencrypt-prod
spec:
  acme:
    email: satyaande2004@gmail.com
    server: https://acme-v02.api.letsencrypt.org/directory
    privateKeySecretRef:
      name: letsencrypt-prod
    solvers:
      - http01:
          ingress:
            class: nginx
			
			


certificate-webapp.yml
=========================
apiVersion: cert-manager.io/v1
kind: Certificate
metadata:
  name: webapp
spec:
  secretName: webapp-secret
  dnsNames:
    - satyaande.xyz
  issuerRef:
    name: letsencrypt-prod
    kind: ClusterIssuer




application domain.yaml
=====================
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: webapp
  annotations:
    kubernetes.io/ingress.class: nginx
spec:
  rules:
    - host: satyaande.xyz
      http:
        paths:
        - pathType: Prefix
          path: /
          backend:
            service:
              name: mywebapp
              port:
                number: 80
  tls:
    - hosts:
        - satyaande.xyz
      secretName: webapp-secret
